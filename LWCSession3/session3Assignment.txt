Javascript:

import SystemModstamp from "@salesforce/schema/Account.SystemModstamp";
import { LightningElement, track } from "lwc";
export default class MyFirstLWCComponent extends LightningElement {
    @track firstName;
    @track lastName;
    submitHandler(event) {
        var inputs=this.template.querySelectorAll("lightning-input");

        inputs.forEach(function(element){
            if(element.name=="FirstName")
            {
                this.firstName=element.value;
                element.disabled = true;
            }

            else if(element.name=="LastName")
            {
                this.lastName=element.value;
                element.disabled=true;
            }   
        },this);

        var button = this.template.querySelector("lightning-button");
        button.disabled = true;
        
        console.log("Welcome "+{firstName}+{lastName});
        alert("Welcome "+{firstName}+{lastName});

        
        


    }
}
